Overview

A payment engine built for scale, resilience, and zero data loss.

PayStream is a distributed real-time payment processing engine that handles over 50,000 transactions per second with end-to-end P99 latency under 80ms. It was built to replace a legacy monolith that was hitting operational limits under growth.

The system is designed around event sourcing and CQRS — every payment is an immutable event in a Kafka-backed stream. This gives us full auditability, easy replay-based debugging, and horizontal scalability without distributed transactions.

I led the backend architecture design and implementation, working closely with the infra and platform teams to ensure the system would be operable by on-call engineers at 3am.

— The challenges

  • Legacy monolith hitting limits at 3k TPS with no clear path to scale horizontally
  • Distributed transactions across multiple databases causing deadlocks and inconsistent states under load
  • No idempotency guarantees — duplicate charges during network retries were a real customer-facing bug
  • Full deploy took 45 minutes; any payment bug required a full rollback cycle
  • Zero audit trail — no way to replay or reconstruct the exact sequence of events for a failed payment
  • Fraud detection ran synchronously in the critical payment path, adding 400ms+ of latency

+ How we solved it

  • Decomposed monolith into stateless microservices — each owns a single bounded context and scales independently
  • Replaced distributed transactions with event sourcing: every state change is a published event, consumed asynchronously
  • Implemented idempotency keys with a Redis-backed deduplication layer — retries are always safe
  • Adopted feature flags + canary deployments; we now deploy 20+ times per day with zero downtime
  • Built a full event ledger in PostgreSQL with TimescaleDB for analytics — any payment can be replayed from raw events
  • Moved fraud detection off the critical path via an async Kafka consumer — fraud decisions arrive in <200ms without blocking payment response

package processor

import (
  "context"
  "fmt"
  "github.com/paystream/internal/events"
  "github.com/paystream/internal/idempotency"
)

// ProcessPayment is the entry point for all payment requests.
// It guarantees exactly-once processing via idempotency keys.
func (s *Service) ProcessPayment(
  ctx context.Context,
  req *PaymentRequest,
) (*PaymentResult, error) {

  // 1. Idempotency check — if we've seen this key, return cached result
  if cached, err := s.idempotency.Get(ctx, req.IdempotencyKey); err == nil {
    s.metrics.IncrCounter("payment.idempotent_hit")
    return cached, nil
  }

  // 2. Validate and enrich the payment request
  enriched, err := s.validator.Enrich(ctx, req)
  if err != nil {
    return nil, fmt.Errorf("validation failed: %w", err)
  }

  // 3. Publish PaymentInitiated event — the source of truth
  evt := events.PaymentInitiated{
    PaymentID:      enriched.ID,
    Amount:         enriched.Amount,
    Currency:       enriched.Currency,
    IdempotencyKey: req.IdempotencyKey,
    Timestamp:      s.clock.Now(),
  }

  if err := s.stream.Publish(ctx, "payments.initiated", evt); err != nil {
    return nil, fmt.Errorf("publish failed: %w", err)
  }

  // 4. Wait for settlement confirmation (or timeout)
  result, err := s.settler.Await(ctx, enriched.ID)
  if err != nil {
    return nil, err
  }

  // 5. Cache result for future idempotent reads
  _ = s.idempotency.Set(ctx, req.IdempotencyKey, result)

  return result, nil
}
internal/idempotency/store.go 
package idempotency

import (
  "context"
  "encoding/json"
  "time"
  "github.com/redis/go-redis/v9"
)

const (
  keyPrefix = "idem:"
  ttl       = 24 * time.Hour
)

// Store is a Redis-backed idempotency key store.
// Keys are namespaced per-merchant to avoid collisions.
type Store struct {
  rdb    *redis.ClusterClient
  prefix string
}

func (s *Store) Set(
  ctx context.Context,
  key string,
  result *PaymentResult,
) error {
  data, err := json.Marshal(result)
  if err != nil {
    return err
  }

  // NX flag: only set if key doesn't exist (race-safe)
  return s.rdb.SetNX(ctx, s.prefix+keyPrefix+key, data, ttl).Err()
}

func (s *Store) Get(
  ctx context.Context,
  key string,
) (*PaymentResult, error) {
  data, err := s.rdb.Get(ctx, s.prefix+keyPrefix+key).Bytes()
  if err != nil {
    return nil, err // redis.Nil if not found
  }

  var result PaymentResult
  return &result, json.Unmarshal(data, &result)
}
internal/events/store.go 
package events

import (
  "context"
  "fmt"
  "time"
  "github.com/jackc/pgx/v5"
)

// Append writes an event to the PostgreSQL event ledger.
// This is the single source of truth for all payment state.
func (s *Store) Append(
  ctx context.Context,
  streamID string,
  event Event,
  expectedVersion int64,
) error {
  const q = `
    INSERT INTO event_store
      (stream_id, event_type, payload, version, occurred_at)
    SELECT $1, $2, $3, $4, $5
    WHERE (
      SELECT COALESCE(MAX(version), 0)
      FROM event_store WHERE stream_id = $1
    ) = $4 - 1
  `

  result, err := s.db.Exec(ctx, q,
    streamID,
    event.Type(),
    event,
    expectedVersion,
    time.Now().UTC(),
  )
  if err != nil {
    return fmt.Errorf("append event: %w", err)
  }

  // Optimistic concurrency: 0 rows = version conflict
  if result.RowsAffected() == 0 {
    return ErrVersionConflict
  }
  return nil
}
